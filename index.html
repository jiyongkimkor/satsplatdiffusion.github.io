const loadTasks = models.map(async (model) => { try { const obj = await SPLAT.Loader.LoadAsync(model.url, scene, (progress) => { status.innerText = `Loading ${model.name}: ${Math.round(progress * 100)}%`; }); // 1. 축 보정 (X축 기준 90도 회전) // 가우시안 왜곡을 방지하기 위해 Quaternion을 직접 계산하여 할당 const rotationAngle = Math.PI / 2; obj.rotation = SPLAT.Quaternion.FromEuler(new SPLAT.Vector3(rotationAngle, 0, 0)); // 2. 중심 계산 및 이동 const data = obj.data || (obj.splat && obj.splat.data); if (data && data.positions) { const positions = data.positions; let avgX = 0, avgY = 0, avgZ = 0; const count = positions.length / 3; for (let i = 0; i < positions.length; i += 3) { avgX += positions[i]; avgY += positions[i + 1]; avgZ += positions[i + 2]; } const center = new SPLAT.Vector3(avgX / count, avgY / count, avgZ / count); // [핵심] 회전이 적용된 상태에서 위치를 보정해야 하므로 // 중심점을 먼저 회전시킨 후 그만큼 이동시킵니다. const rotatedCenter = center.rotate(obj.rotation); obj.position = rotatedCenter.multiply(-1); } return obj; } catch (err) { console.error(`Failed to load ${model.name}:`, err); return null; } });
